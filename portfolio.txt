<script>
  (function(){
    const OVERLAY_ID = 'promo-modal-overlay';
    const DONT_SHOW_KEY = 'prabhu_promo_dont_show';
    const overlay = document.getElementById(OVERLAY_ID);
    const closeBtn = document.getElementById('promo-close');
    const dontShow = document.getElementById('promo-dontshow');
    const viewBtn = document.getElementById('promo-view');
    const promoImage = document.getElementById('promo-image');

    // Do not show if user previously opted out
    try {
      const optedOut = localStorage.getItem(DONT_SHOW_KEY);
      if(optedOut === 'true') return;
    } catch(e){ /* localStorage may be unavailable, continue to show */ }

    // Show modal after DOM is ready
    function showModal(){
      overlay.style.display = 'flex';
      overlay.setAttribute('aria-hidden','false');
      // trap focus on modal for accessibility
      const focusable = overlay.querySelectorAll('a,button,input');
      if(focusable.length) focusable[0].focus();
      document.body.style.overflow = 'hidden';
    }

    function hideModal(){
      overlay.style.display = 'none';
      overlay.setAttribute('aria-hidden','true');
      document.body.style.overflow = '';
      // store preference if checked
      try {
        if(dontShow.checked) localStorage.setItem(DONT_SHOW_KEY, 'true');
      } catch(e){}
    }

    // Open lightbox using existing openLightbox if available
    function openPromoLightbox(){
      const src = promoImage.getAttribute('data-src') || promoImage.src;
      if(typeof openLightbox === 'function'){
        openLightbox('image', src);
        hideModal();
      } else {
        // fallback: open image in new tab
        window.open(src, '_blank');
        hideModal();
      }
    }

    // Event listeners
    closeBtn.addEventListener('click', hideModal);
    overlay.addEventListener('click', function(e){
      if(e.target === overlay) hideModal();
    });
    document.addEventListener('keydown', function(e){
      if(e.key === 'Escape' && overlay.style.display === 'flex') hideModal();
    });

    viewBtn.addEventListener('click', openPromoLightbox);
    promoImage.addEventListener('click', openPromoLightbox);

    // Make call and WhatsApp buttons keyboard friendly
    ['promo-call','promo-wa'].forEach(id => {
      const el = document.getElementById(id);
      if(!el) return;
      el.addEventListener('keydown', function(e){
        if(e.key === 'Enter' || e.key === ' ') { e.preventDefault(); el.click(); }
      });
    });

    // Delay showing slightly so page content loads first
    window.addEventListener('load', function(){
      // small delay to avoid jarring on very slow pages
      setTimeout(showModal, 450);
    });
  })();
</script>
